import { Input } from 'minerva-ui';

Extends native `<input />` component

```jsx
import { Input } from 'minerva-ui';
```

```jsx live
() => {
  const [name, setName] = React.useState('');
  return (
    <Input
      value={name}
      onChange={e => setName(e.target.value)}
      placeholder="Basic Input"
    />
  );
};
```

# Examples

### Disabled Input

```jsx live
() => {
  const [name, setName] = React.useState('');
  return (
    <Input
      disabled
      value={name}
      onChange={e => setName(e.target.value)}
      placeholder="Basic Input"
    />
  );
};
```

```jsx live
      type="password"

```
